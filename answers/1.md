var twoSum = function(nums, target) {
  var result = [];
  var map = {};
  var temp;
  for (var i = 0, len = nums.length; i < len; i++) {
      temp = target - nums[i];
      if (map[temp] !== undefined) {
          result[0] = parseInt(map[temp], 10) ;
          result[1] = i;
          return result;
      }
      map[nums[i]] = i;
  }
  return -1;
};

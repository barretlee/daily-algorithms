function reverse (number) {
    const min = - ((1 << 30) * 2);
    const max = (1 << 30) * 2 - 1;
    let result;
    const numberArr = number.toString().split('');
    if(number < 0) {
        numberArr.shift();
    }
    result = Number(numberArr.reverse().join(''));
    if(number < 0) {
        result *= -1;
    }
    if(result < min || result > max) {
        result = 0;
    }

    return result;
}
